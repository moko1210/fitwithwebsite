### パフォーマンステストスクリプト

```javascript
// performance-test.js
async function runPerformanceTests() {
  const urls = [
    'index.html',
    'product.html',
    'mission.html',
    'news.html',
    'contact.html'
  ];

  const results = [];

  for (const url of urls) {
    console.log(`Testing: ${url}`);

    // ページロード
    const startTime = performance.now();
    await fetch(url);
    const endTime = performance.now();

    results.push({
      url,
      loadTime: endTime - startTime
    });
  }

  console.table(results);

  // 目標: すべてのページが2秒以内
  const failed = results.filter(r => r.loadTime > 2000);
  if (failed.length > 0) {
    console.error('以下のページが目標を達成できませんでした:', failed);
  } else {
    console.log('✓ すべてのページがパフォーマンス目標を達成！');
  }
}
```
